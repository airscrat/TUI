102  //returns distance between center of 'a' and edge of 'b'
103  //if center of 'a' intersects with 'b' return value is negative
104  auto dst = [&](surface* a, surface* b)
105  {
106  const vec2f ca = center(a);
107  const vec2f cb = center(b);
108  const vec2i sb = b->getSize();
109 
110  vec2f d = {
111  std::fabs(ca.x - cb.x) - sb.x/2.f,
112  std::fabs(ca.y - cb.y) - sb.y/2.f
113  };
114  const bool intersect = d.x < 0 && d.y < 0;
115 
116  if (!intersect)
117  {
118  d.x = std::max(d.x, 0.f);
119  d.y = std::max(d.y, 0.f);
120  return std::sqrt(d.x * d.x + d.y * d.y);
121  }
122  else
123  {
124  return std::max(d.x, d.y);
125  }
126  };
