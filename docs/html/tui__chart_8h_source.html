140  void fill()

142  clear();
143  makeBlank();

153  updateSurfaceSize(m_chart);
154  m_chart.makeTransparent();
155 
156  if (m_display_value_labels)
157  {
158  for (int i = 0; i < m_max_str.size(); i++)
159  {
160  setSymbolAt(m_max_str[i], { i,0 });
161  }
162  for (int i = 0; i < m_min_str.size(); i++)
163  {
164  int y_pos = std::max(0, m_chart.getSize().y - 1 - m_display_data_labels);
165  setSymbolAt(m_min_str[i], { i,y_pos });
166  }
167  }
168 
169  float distance = (fabs(m_min - m_max));
170  if (m_min > 0) { distance += m_min; }
171  if (m_max < 0) { distance += fabs(m_max); }
172 
173  if (distance > 0)
174  {
175  int halves = (m_chart.getSize().y - m_display_data_labels) * 2;
176  int p_halves = round(m_max / distance * halves) * (m_max>=0);
177 
178  int scroll_pos = m_scroll.getTopPosition();
179  int x = m_distance *(scroll_pos % m_distance != 0) - scroll_pos % m_distance;
180 
181  for (int i = ceil(scroll_pos / (float)m_distance); (i < m_values.size() && x < m_chart.getSize().x); i++, x += m_distance)
182  {
183  int h = round(fabs(m_values[i].value) / distance * halves);
184 
185  auto isFull = [&](int y)
186  {
187  switch (m_values[i].value >= 0)
188  {
189  case true:
190  return y >= p_halves - h && y < p_halves;
191  case false:
192  return y >= p_halves && y < p_halves + h;
193  }
194  };
195 
196  for (int y = 0; y < halves; y+=2)
197  {
198  if (isFull(y) && isFull(y + 1))
199  {
200  m_chart.setSymbolAt(gca().full, { x, y / 2 });
201  }
202  else if (isFull(y) && !isFull(y + 1))
203  {
204  m_chart.setSymbolAt(gca().upper_half, { x, y / 2 });
205  }
206  else if (!isFull(y) && isFull(y + 1))
207  {
208  m_chart.setSymbolAt(gca().lower_half, { x, y / 2 });
209  }
210  }
211 
212  if (m_display_data_labels)
213  {
214  symbol_string fw_name = getFullWidthString(m_values[i].name);
215 
216  for (int j = 0; j < fw_name.size() && x + j < m_chart.getSize().x; j++)
217  {
218  m_chart.setSymbolAt(fw_name[j], { x + j, m_chart.getSize().y - 1 });
219  }
220  }
221  }
222  }
